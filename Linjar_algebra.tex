\section*{Kapitel 1: Linjära ekvationssystem}
\texttt{\tskcol{1.1~~~~~ (s.)}}
\begin{quotation}
	\noindent
	Börja nerifrån och upp och lös en variabel i taget.
	\begin{align*}
	&\systeme{
		2x+3y- z=5,
		  -3y+5z=1,
		      4z=8}  \\ \Leftrightarrow
	&\begin{cases}
	z=2 \\
	y=\frac{1-5*2}{-3}=3 \\
	x=\frac{5+2-3*3}{2}=-1
	\end{cases}
	\end{align*}
	\\
	\textbf{Svar}: $(x,y,z)=(-1,3,2)$
\end{quotation}

\texttt{\tskcol{1.2~~~~~ (s.)}}
\begin{quotation}
	\noindent
	Gausselimination
	\begin{align*}
	&\systeme{
		  x-2y+  z=2,
		 2x-6y+11z=35,
		-3x+5y+  z=8}  
	&\begin{array}{l} 
		(a) \\ 
		(b) \\
		(c)
	\end{array} \\ \Leftrightarrow
	&\systeme{
		x-2y+ z=2,
		 -2y+9z=31,
		 - y+4z=14}  
	&\begin{array}{l} 
	(a')=(a) \\ 
	(b')=(b)-2(a) \\
	(c')=(c)+3(a)
	\end{array} \\ \Leftrightarrow
	&\systeme{
		x-2y+   z=2,
		 -2y+  9z=31,
		    -0.5z=-1.5}  
	&\begin{array}{l} 
	(a'')=(a') \\ 
	(b'')=(b') \\
	(c'')=(c')-0.5(b')
	\end{array} \\ \Leftrightarrow
	&\begin{cases}
	z=3 \\
	y=\frac{31-9*3}{-2} =-2\\
	x=2+2*(-2)-3=-5
	\end{cases}
	\end{align*}
	\\
	\textbf{Svar}: $(x,y,z)=(-5,-2,3)$
\end{quotation}
